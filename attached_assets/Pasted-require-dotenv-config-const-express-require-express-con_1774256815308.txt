require("dotenv").config();

const express = require("express");
const axios = require("axios");
const sqlite3 = require("sqlite3").verbose();
const cors = require("cors");
const session = require("express-session");
const bcrypt = require("bcrypt");
const helmet = require("helmet");
const rateLimit = require("express-rate-limit");
const { body, validationResult } = require("express-validator");
const { v4: uuidv4 } = require("uuid");

const app = express();

// --- SECURITY ---
app.use(helmet());
app.use(express.json());

app.use(rateLimit({
  windowMs: 60000,
  max: 100
}));

app.use(cors({
  origin: ["http://localhost:3000"],
  credentials: true
}));

app.use(session({
  secret: process.env.SESSION_SECRET,
  resave: false,
  saveUninitialized: false,
  cookie: { httpOnly: true, sameSite: "lax" }
}));

app.use(express.static("public"));

// --- DB ---
const db = new sqlite3.Database("./db.sqlite");

// USERS
db.run(`
CREATE TABLE IF NOT EXISTS users (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  email TEXT UNIQUE,
  username TEXT,
  password TEXT,
  uid TEXT,
  avatar TEXT
)
`);

// ORDERS
db.run(`
CREATE TABLE IF NOT EXISTS orders (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  phone TEXT,
  network TEXT,
  expectedAmount REAL,
  status TEXT,
  txid TEXT,
  confirmations INTEGER DEFAULT 0,
  userId INTEGER,
  depositId TEXT,
  paid INTEGER DEFAULT 0,
  createdAt DATETIME DEFAULT CURRENT_TIMESTAMP
)
`);

// --- SIGNUP ---
app.post("/signup",
  body("email").isEmail(),
  body("password").isLength({ min: 6 }),
  async (req, res) => {

    const errors = validationResult(req);
    if (!errors.isEmpty()) return res.json({ error: "Invalid input" });

    const { email, username, password } = req.body;

    const hash = await bcrypt.hash(password, 10);
    const uid = "MB" + Math.floor(100000 + Math.random() * 900000);
    const avatar = `https://api.dicebear.com/7.x/initials/svg?seed=${username}`;

    db.run(
      `INSERT INTO users (email, username, password, uid, avatar)
       VALUES (?, ?, ?, ?, ?)`,
      [email, username, hash, uid, avatar],
      function (err) {
        if (err) return res.json({ error: "User exists" });

        req.session.userId = this.lastID;
        res.json({ success: true });
      }
    );
  }
);

// --- LOGIN ---
const loginLimiter = rateLimit({ windowMs: 15*60*1000, max: 5 });

app.post("/login", loginLimiter, (req, res) => {
  const { email, password } = req.body;

  db.get("SELECT * FROM users WHERE email=?", [email], async (err, user) => {
    if (!user) return res.json({ error: "User not found" });

    const match = await bcrypt.compare(password, user.password);
    if (!match) return res.json({ error: "Wrong password" });

    req.session.userId = user.id;
    res.json({ success: true });
  });
});

// --- SESSION USER ---
app.get("/me", (req, res) => {
  if (!req.session.userId) return res.json(null);

  db.get("SELECT id, username, uid FROM users WHERE id=?",
    [req.session.userId],
    (err, user) => res.json(user)
  );
});

// --- CREATE ORDER ---
app.post("/create-order", (req, res) => {
  if (!req.session.userId) return res.json({ error: "Unauthorized" });

  const { phone, network, amount } = req.body;
  const depositId = uuidv4();

  db.run(
    `INSERT INTO orders 
    (phone, network, expectedAmount, status, userId, depositId)
    VALUES (?, ?, ?, ?, ?, ?)`,
    [phone, network, amount, "waiting", req.session.userId, depositId],
    function () {
      res.json({
        orderId: this.lastID,
        address: process.env.WALLET_ADDRESS,
        memo: depositId
      });
    }
  );
});

// --- USER ORDERS ---
app.get("/my-orders", (req, res) => {
  if (!req.session.userId) return res.json([]);

  db.all(
    "SELECT * FROM orders WHERE userId=? ORDER BY id DESC",
    [req.session.userId],
    (err, rows) => res.json(rows)
  );
});

// --- WALLET WATCHER ---
async function watchWallet() {
  try {
    const res = await axios.get(
      `${process.env.TRON_API}/v1/accounts/${process.env.WALLET_ADDRESS}/transactions/trc20`
    );

    for (let tx of res.data.data) {

      if (tx.token_info.address !== process.env.USDT_CONTRACT) continue;

      const amount = parseInt(tx.value) / 1e6;
      const confirmations = tx.block_timestamp ? 1 : 0;

      db.get(
        `SELECT * FROM orders 
         WHERE status='waiting'
         AND ABS(expectedAmount - ?) < 1
         AND paid=0`,
        [amount],
        async (err, order) => {
          if (!order) return;
          if (confirmations < 1) return;

          db.run(
            "UPDATE orders SET paid=1, status='processing', txid=?, confirmations=? WHERE id=?",
            [tx.transaction_id, confirmations, order.id]
          );

          await executePayout(order, amount);
        }
      );
    }
  } catch (e) {
    console.log("Watcher error:", e.message);
  }
}

setInterval(watchWallet, 10000);

// --- PAYOUT ---
async function executePayout(order, amount) {
  try {
    const ugx = amount * 3700;

    await axios.post(
      "https://api.flutterwave.com/v3/transfers",
      {
        account_bank: order.network,
        account_number: order.phone,
        amount: ugx,
        currency: "UGX"
      },
      {
        headers: {
          Authorization: `Bearer ${process.env.FLW_SECRET_KEY}`
        }
      }
    );

    db.run("UPDATE orders SET status='completed' WHERE id=?", [order.id]);

  } catch (err) {
    db.run("UPDATE orders SET status='failed' WHERE id=?", [order.id]);
  }
}

app.listen(process.env.PORT, () => {
  console.log("MBIO PAY running on", process.env.PORT);
});
