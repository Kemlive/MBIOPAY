require("dotenv").config();
const express = require("express");
const axios = require("axios");
const sqlite3 = require("sqlite3").verbose();
const cors = require("cors");
const path = require("path");

const app = express();
app.use(express.json());
app.use(cors());
app.use(express.static("public"));

const db = new sqlite3.Database("./db.sqlite");

// --- DB TABLE ---
db.run(`
CREATE TABLE IF NOT EXISTS orders (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  phone TEXT,
  network TEXT,
  amount REAL,
  status TEXT,
  txid TEXT
)
`);

// --- CREATE ORDER ---
app.post("/create-order", (req, res) => {
  const { phone, network } = req.body;

  db.run(
    "INSERT INTO orders (phone, network, status) VALUES (?, ?, ?)",
    [phone, network, "waiting"],
    function (err) {
      res.json({
        orderId: this.lastID,
        address: process.env.WALLET_ADDRESS
      });
    }
  );
});

// --- GET ORDER ---
app.get("/order/:id", (req, res) => {
  db.get("SELECT * FROM orders WHERE id=?", [req.params.id], (err, row) => {
    res.json(row);
  });
});

// --- WALLET WATCHER ---
let lastTx = "";

async function watchWallet() {
  try {
    const res = await axios.get(
      `${process.env.TRON_API}/v1/accounts/${process.env.WALLET_ADDRESS}/transactions/trc20`
    );

    const txs = res.data.data;

    for (let tx of txs) {
      if (tx.transaction_id === lastTx) continue;

      if (tx.to === process.env.WALLET_ADDRESS) {
        lastTx = tx.transaction_id;

        const amount = parseInt(tx.value) / 1e6;

        db.get(
          "SELECT * FROM orders WHERE status='waiting' LIMIT 1",
          async (err, order) => {
            if (!order) return;

            db.run(
              "UPDATE orders SET status=?, amount=?, txid=? WHERE id=?",
              ["processing", amount, tx.transaction_id, order.id]
            );

            await executePayout(order, amount);
          }
        );
      }
    }
  } catch (e) {
    console.log("Watcher error:", e.message);
  }
}

setInterval(watchWallet, 10000);

// --- EXECUTION ENGINE ---
async function executePayout(order, amount) {
  try {
    const ugx = amount * 3700;

    await axios.post(
      "https://api.flutterwave.com/v3/transfers",
      {
        account_bank: order.network,
        account_number: order.phone,
        amount: ugx,
        currency: "UGX",
        narration: "Crypto payout"
      },
      {
        headers: {
          Authorization: `Bearer ${process.env.FLW_SECRET_KEY}`
        }
      }
    );

    db.run("UPDATE orders SET status=? WHERE id=?", [
      "completed",
      order.id
    ]);

    console.log("Payout success:", order.id);

  } catch (err) {
    console.log("Payout failed:", err.message);

    db.run("UPDATE orders SET status=? WHERE id=?", [
      "failed",
      order.id
    ]);
  }
}

// --- START SERVER ---
app.listen(process.env.PORT, () => {
  console.log("Server running on port", process.env.PORT);
});
