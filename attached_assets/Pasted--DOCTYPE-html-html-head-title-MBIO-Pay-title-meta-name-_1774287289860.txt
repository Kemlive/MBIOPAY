<div id="app"></div>

<script>

// ===== STATE =====
let state = {
  step: "onboarding",
  phone: "",
  network: "MTN",
  amount: 0,
  orderId: null,
  address: "",
  countdown: 1800
};

// ===== ROUTER =====
function render() {
  if (state.step === "onboarding") return onboarding();
  if (state.step === "send") return send();
  if (state.step === "confirm") return confirm();
  if (state.step === "pay") return pay();
  if (state.step === "success") return success();
}

// ===== STEP 1 =====
function onboarding() {
  document.getElementById("app").innerHTML = `
    <div class="center">
      <h2>MBIO Pay</h2>
      <p>Send USDT → Mobile Money instantly</p>
      <button onclick="go('send')">Start in 10 seconds</button>
    </div>
  `;
}

// ===== STEP 2 =====
function send() {
  document.getElementById("app").innerHTML = `
    <div class="card">
      <h3>Send Money</h3>

      <input id="phone" placeholder="256..." />
      
      <select id="network">
        <option>MTN</option>
        <option>Airtel</option>
      </select>

      <input id="amount" placeholder="Amount USDT" />

      <p class="muted">≈ You'll receive UGX (estimated)</p>

      <button onclick="goConfirm()">Continue</button>
    </div>
  `;
}

function goConfirm() {
  state.phone = document.getElementById("phone").value;
  state.network = document.getElementById("network").value;
  state.amount = parseFloat(document.getElementById("amount").value);

  state.step = "confirm";
  render();
}

// ===== STEP 3 =====
function confirm() {

  const fee = state.amount * 0.01;
  const rate = 3550;
  const final = (state.amount - fee) * rate;

  document.getElementById("app").innerHTML = `
    <div class="card">
      <h3>Confirm Transfer</h3>

      <p><b>Recipient:</b> ${state.phone} (${state.network})</p>
      <p>✔ Verified Recipient</p>

      <hr/>

      <p>You Send: ${state.amount} USDT</p>
      <p>Fee: ${fee.toFixed(2)} USDT</p>
      <p>Rate: ${rate} UGX</p>

      <p class="big">Recipient Gets: ${Math.floor(final)} UGX</p>

      <p>⏳ Rate locked for <span id="timer"></span></p>

      <p class="muted">
        ✔ Funds protected <br/>
        ✔ Refund if failed <br/>
        ✔ Arrives in ~2 minutes
      </p>

      <button onclick="createOrder()">🚀 Send Now</button>
    </div>
  `;

  startTimer();
}

// ===== TIMER =====
function startTimer() {
  const interval = setInterval(() => {
    if (state.countdown <= 0) {
      clearInterval(interval);
      return;
    }

    state.countdown--;

    const min = Math.floor(state.countdown / 60);
    const sec = state.countdown % 60;

    document.getElementById("timer").innerText = `${min}:${sec}`;
  }, 1000);
}

// ===== CREATE ORDER =====
async function createOrder() {

  const res = await fetch("/create-order", {
    method: "POST",
    headers: {"Content-Type":"application/json"},
    body: JSON.stringify({
      phone: state.phone,
      network: state.network
    })
  });

  const data = await res.json();

  state.orderId = data.orderId;
  state.address = data.address;
  state.step = "pay";

  render();
}

// ===== STEP 4 =====
function pay() {

  document.getElementById("app").innerHTML = `
    <div class="card center">
      <h3>Send USDT</h3>

      <p>Send to:</p>
      <p><b>${state.address}</b></p>

      <button onclick="copy()">Copy Address</button>

      <p class="muted">Waiting for payment...</p>

      <div id="status"></div>
    </div>
  `;

  watchStatus();
}

function copy() {
  navigator.clipboard.writeText(state.address);
  alert("Copied");
}

// ===== WATCH PAYMENT =====
function watchStatus() {

  const interval = setInterval(async () => {

    const res = await fetch("/order/" + state.orderId);
    const data = await res.json();

    document.getElementById("status").innerHTML =
      "<b>Status:</b> " + data.status;

    if (data.status === "completed") {
      clearInterval(interval);
      state.step = "success";
      render();
    }

  }, 5000);
}

// ===== STEP 5 =====
function success() {

  document.getElementById("app").innerHTML = `
    <div class="card center">
      <h2>✅ Success</h2>

      <p>Money sent successfully</p>

      <p class="muted">Users sent UGX today</p>

      <button onclick="go('send')">Send Again</button>
    </div>
  `;
}

// ===== NAVIGATION =====
function go(step) {
  state.step = step;
  render();
}

// INIT
render();

</script>
