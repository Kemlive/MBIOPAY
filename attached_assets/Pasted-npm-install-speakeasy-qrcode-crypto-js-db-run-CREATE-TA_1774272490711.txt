const speakeasy = require("speakeasy");
const QRCode = require("qrcode");

app.get("/admin/setup", async (req, res) => {

  const secret = speakeasy.generateSecret({ length: 20 });

  const qr = await QRCode.toDataURL(secret.otpauth_url);

  res.json({
    qr,
    base32: secret.base32
  });
});

const bcrypt = require("bcrypt");

app.post("/admin/register", async (req, res) => {

  const { email, password, totpSecret } = req.body;

  const hash = await bcrypt.hash(password, 10);

  db.run(
    `INSERT INTO admins (email,password,totpSecret,createdAt)
     VALUES (?,?,?,?)`,
    [email, hash, totpSecret, Date.now()],
    () => res.json({ ok: true })
  );
});

<script>
function getDeviceFingerprint() {
  return btoa(
    navigator.userAgent +
    screen.width +
    screen.height +
    navigator.language
  );
}
</script>

const cryptoJS = require("crypto-js");

app.post("/admin/login", async (req, res) => {

  const { email, password, token, device } = req.body;

  db.get("SELECT * FROM admins WHERE email=?", [email], async (e, admin) => {

    if (!admin) return res.json({ err: "no admin" });

    const passOk = await bcrypt.compare(password, admin.password);
    if (!passOk) return res.json({ err: "bad password" });

    // 🔐 VERIFY TOTP
    const verified = speakeasy.totp.verify({
      secret: admin.totpSecret,
      encoding: "base32",
      token,
      window: 1
    });

    if (!verified) return res.json({ err: "invalid OTP" });

    // 🔒 DEVICE CHECK
    const deviceHash = cryptoJS.SHA256(device).toString();

    if (admin.deviceHash && admin.deviceHash !== deviceHash) {
      return res.json({ err: "New device blocked" });
    }

    // First login → bind device
    if (!admin.deviceHash) {
      db.run(
        "UPDATE admins SET deviceHash=? WHERE id=?",
        [deviceHash, admin.id]
      );
    }

    // SESSION
    req.session.isAdmin = true;
    req.session.adminId = admin.id;

    res.json({ ok: true });
  });
});

function isAdmin(req, res, next) {
  if (!req.session.isAdmin) {
    return res.status(403).json({ error: "Admin only" });
  }
  next();
}

async function login() {

  const device = getDeviceFingerprint();
